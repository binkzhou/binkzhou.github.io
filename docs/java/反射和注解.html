16.遍历方法名

public class Demo {
    @Test
    public void tese() throws Exception {
        //1.获取Person对象
        Class cs = Class.forName("com.zh.model.Person");
        //2.获取所有public修饰的方法
        Method[] methods = cs.getMethods();
        for (Method method : methods) {
            //3.获取方法名getName()
            System.out.println(method.getName());
        }
        System.out.println("--------------------");
        //4.获取全类名
        String className = cs.getName();
        System.out.println(className);
    }
}

8.元注解@Target注解和作用位置

/*
元注解：用于描述注解的注解
 @Target:描述注解能够作用的位置
 @Retention: 描述注解被保留的阶段 
 @Documented: 描述注解是否被抽取到api文档中
 @Inherited：描述注解是否被子类继承
*/
@Target(value = {ElementType.TYPE}) //只能作用到类上
public @interface MyAnno3 {
}
//源码
public enum ElementType {
    /* 作用于类上 */
    TYPE,
    /* 作用于成员变量上*/
    FIELD,
    /* 作用于方法上 */
    METHOD,
    PARAMETER,
    CONSTRUCTOR,
    LOCAL_VARIABLE,
    ANNOTATION_TYPE,
    PACKAGE,
    
    TYPE_PARAMETER,
    
    TYPE_USE
}

13.注解的解析

//1.定义注解
@Retention(RetentionPolicy.RUNTIME)
public @interface Person {
    int id();
    String name();
    int age();
}
/**********************************************/
//2.使用注解
@Person(id = 1,name = "小明",age = 15)
public class Student {
}
/**********************************************/
//3.解析注解
public class test {
    public static void main(String[] args) {
        //获取该类字节码对象
        Class<Student> studentClass = Student.class;
        //获取上面的注解对象
        Person annotation = studentClass.getAnnotation(Person.class);
        //调用注解抽象方法获取返回值
        int id = annotation.id();
        String name = annotation.name();
        int age = annotation.age();
        System.out.println("id:"+id);
        System.out.println("name:"+name);
        System.out.println("age:"+age);
    }
}
/**********************************************/
/*
输出
id:1
name:小明
age:15
*/
